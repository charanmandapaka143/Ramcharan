# Read number of elements
n = int(input("Enter number of elements: "))

# Create empty array
arr = []

# Read elements into array
for i in range(n):
    value = int(input(f"Enter element {i + 1}: "))
    arr.append(value)

# Initialize values
total = 0
largest = arr[0]
smallest = arr[0]

# Iterate through array
for num in arr:
    total += num

    if num > largest:
        largest = num

    if num < smallest:
        smallest = num

# Calculate average
average = total / n
